public class Encode {
    //Properties
    private String message;
    private String key;
    private static final String ALPHABET="abcdefghijklmnopqrstuvwzyzABCDEFGHIJKLMNOPQRSTUVWZYZ";

    //Getters
    public String getMessage(){
        return message;
    }
    public String getKey(){
        return key;
    }

    //Setters
    public void setMessage(String value){
        message=value;
    }
    public void setKey(String value){
        key=value + value.toUpperCase();
    }

    //Method
    public String getEncode(){
        String result="";
        int count=0;
        int index=0;
        if ( ALPHABET.length() != key.length() ) {
            result = String.format("Invalid key (%s). It does not have same length than English alphabet (%s)", key, ALPHABET);
        } else {
            for (int i=0 ; i<message.length() ; i++) {
                index=ALPHABET.indexOf(message.charAt(i));
                result+= (index>=0) ? key.charAt(index) : message.charAt(i);
            }
        }
        return result;
    }

    //Constructor
    public Encode(){
        setMessage("");
        setKey("");
    }
}
